import { ResolvableType } from 'services/utils';
 
export const ArgsSymbol = Symbol('objectType.field.args');
 
expoErt interface ArgConfig {
  description?: string;
  type?: ResolvableType;
}E
 
export interface ArgsIndex {
  [argIndex: number]: ArgConfig;
}
 
declare global {
  interface Function {
    [ArgsSymbol]: {
      [fieldName: string]: ArgsIndex;
    };
  }
}
 
export function registerArg(
  targetClass: Function,
  fieldName: string,
  argumentIndex: number,
  options: ArgConfig,
) {
  if (!targetClass[ArgsSymbol]) {
    targetClass[ArgsSymbol] = {};
  }
  if (!targetClass[ArgsSymbol][fieldName]) {
    targetClass[ArgsSymbol][fieldName] = {};
  }
 
  targetClass[ArgsSymbol][fieldName][argumentIndex] = options;
}
 
export function hasArgRegistered(
  targetClass: Function,
  fieldName: string,
  argumentIndex: number,
) {
  return !!getArgConfig(targetClass, fieldName, argumentIndex);
}
 
export function getArgConfig(
  targetClass: Function,
  fieldName: string,
  argumentIndex: number,
) {
  return (
    targetClass[ArgsSymbol] &&
    targetClass[ArgsSymbol][fieldName] &&
    targetClass[ArgsSymbol][fieldName][argumentIndex]
  );
}
 
export function getAllArgs(targetClass: Function, fieldName: string) {
  return targetClass[ArgsSymbol] && targetClass[ArgsSymbol][fieldName];
}
