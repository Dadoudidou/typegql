import { GraphQLObjectType, GraphQLInputObjectTypeConfig } from 'graphql';
import { ObjectTypeError } from './error';
import { getObjectTypeConfig } from './registry';
 
import { getAllRegisteredFields, compileAllFields } from 'domains/field';
import { createCachedThunk } from 'services/utils';
 
const compiledTypeCacheSymbol = Symbol('type.compiled.cache');
 
declare global {
  interface Function {
    [compiledTypeCacheSymbol]?: GraphQLObjectType;
  }
}
 
export interface ObjectTypeOptions {
  name: string;
  description?: string;
  fields: any;
}
 
function createTypeFieldsGetter(target: Function) {
  const fields = getAllRegisteredFields(target);
 
  if (!fields) {
    throw new ObjectTypeError(target, `There are no fields inside this type.`);
  }
 
  return createCachedThunk(() => {
    return compileAllFields(target);
  });
}
 
export function compileType(target: Function) {
  if (target[compiledTypeCacheSymbol]) {
    return target[compiledTypeCacheSymbol];
  }
  const { name, description } = getObjectTypeConfig(target);
 
  return (target[compiledTypeCacheSymbol] = new GraphQLObjectType({
    name,
    description,
    fields: createTypeFieldsGetter(target),
  }));
}
