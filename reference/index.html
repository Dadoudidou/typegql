@Field

 interface FieldOptions {
  name?: string;
  description?: string;
  type?: any | () => any;
  isNullable?: boolean;
}

@Field(options?: FieldOptions)

interface InputObjectTypeOptions {
  name?: string;

 interface InputFieldOptions {
  description?: string;
  defaultValue?: any;

  name?: string;
  isNullable?: boolean;
